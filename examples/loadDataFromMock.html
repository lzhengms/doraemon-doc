code


import fetch from 'isomorphic-fetch';

document.getElementById('__exampleDom').innerHTML = 'loading...';

fetch('/something')
  .then(res => {
    if (res.status >= 400) {
      throw new Error("Bad response from server");
    }

    res.json().then(json => {
      document.getElementById('__exampleDom').innerHTML = json.result;
    });
  });

desc
